1 Introduction

This is data from a mouse stomach samples of mouse experiments (H.pylori infection, IFNy treatment) hybridized as dual channel libraries to Agilent 014868 Whole genome mouse 4x44 micro arrays.

2 DGE

We here use standard two-channel analysis of the microarray data and compare: - gastric epithelial cells from mice infected with H.pylori and controls - IFN treated vs. control epithelial organoids

all_results = list()
controls = MA$genes$ControlType!=0

MA.avg <- avereps(MA[!controls,], ID=MA[!controls,]$genes$ProbeName)

dswp = ed$dye_swap
cy3_label = ed$Cy3
cy5_label = ed$Cy5
scan_id = ed$Chip.Barcode
ed$array_label = paste(ifelse(dswp, cy3_label, cy5_label), "_",ifelse(dswp,cy5_label, cy3_label),"_",dswp,"_",ed$Scan.ID,sep="")

sel_samples = rownames(subset(ed, Experiment=="IFN"))
ed_tmp = ed[sel_samples,]
design = ifelse(ed_tmp$Cy5_treatment=="IFN", 1, -1)
fit <- lmFit(MA.avg[,sel_samples], design)
fit <- eBayes(fit)
res = topTable(fit, adjust = "BH",number=nrow(fit), confint = T)
ratio_tab = sweep(MA.avg[,sel_samples]$M, 2, design, "*")
colnames(ratio_tab) = ed_tmp$array_label
res = merge(res, ratio_tab, all.x=T, by.x="ProbeName", by.y=0, sort=F)
all_results[["IFN_vs_ctrl_organoids"]] = res

sel_samples = rownames(subset(ed, Experiment=="Infection Antrum"))
ed_tmp = ed[sel_samples,]
design = ifelse(ed_tmp$Cy5_infection=="Hpy", 1, -1)
fit <- lmFit(MA.avg[,sel_samples], design)
fit <- eBayes(fit)
res = topTable(fit, adjust = "BH",number=nrow(fit), confint = T)
ratio_tab = sweep(MA.avg[,sel_samples]$M, 2, design, "*")
colnames(ratio_tab) = ed_tmp$array_label
res = merge(res, ratio_tab, all.x=T, by.x="ProbeName", by.y=0, sort=F)
all_results[["Hpy_vs_NI"]] = res

common_colnames = colnames(topTable(fit, adjust = "BH",number=nrow(fit)))
